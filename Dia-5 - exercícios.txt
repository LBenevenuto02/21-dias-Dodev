Exercício: 1 
let num1 = 0 
let num2 = 0 
let operacao = "Soma; Subtração; Multiplicação; Divisão"


num1 = prompt("Insira um número: ")
num2 = prompt("Insira outro número: ")
operacao = prompt("Desses números, escolha as opções: Soma; Subtração; Multiplicação; Divisão")

switch(operacao){
    case "Soma":
        console.log("A soma é: " + [num1 + num2]);
        break;
    case "Subtração":
        console.log("A subtração é: " + [num1 - num2]);
        break;
    case "Multiplicação":
        console.log("A multiplicação é: " + [num1 * num2]);
        break;
    case "Divisão":
        console.log("A divisão é: " + [num1/num2]);
        break;
    default:
        console.log("Opção inválida")
        break;
}
